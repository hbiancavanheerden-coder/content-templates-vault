        let allTemplates = [];

        let activeTemplate = null;

        let activeFormat = 'all';


        // Clean database formatting tags, entity escapes, and encoding anomalies

        function cleanBodyText(htmlStr) {

            let txt = document.createElement("textarea");

            txt.innerHTML = htmlStr;

            let decoded = txt.value;

            

            decoded = decoded.replace(/<\/p>/g, '\n').replace(/<\/div>/g, '\n').replace(/<br\s*\/?>/g, '\n');

            decoded = decoded.replace(/<[^>]+>/g, '');

            

            decoded = decoded.replace(/â€™/g, "'")

                             .replace(/â€œ/g, '"')

                             .replace(/â€ /g, '"')

                             .replace(/â€”/g, '—')

                             .replace(/â€¢/g, '•')

                             .replace(/&nbsp;/g, ' ');

            

            return decoded.replace(/\n{3,}/g, '\n\n').trim();

        }


        // Auto-discover template placeholder variables via pattern matching

        function extractPlaceholders(text) {

            const regex = /\{([^}]+)\}|\[([^\]]+)\]/g;

            const matches = [];

            let match;

            while ((match = regex.exec(text)) !== null) {

                const placeholder = match[1] || match[2];

                if (!matches.includes(placeholder)) {

                    matches.push(placeholder);

                }

            }

            return matches;

        }


        // UPGRADED MULTI-VAULT LOAD ROUTINE

        // Reads from templates.html plus templates_1.html up to templates_10.html dynamically.

        async function loadVaultData() {

            // Define your distributed data pack files list to reach 500+ items cleanly

            const filesToFetch = [

                './templates.html',

                './templates_1.html',

                './templates_2.html',

                './templates_3.html',

                './templates_4.html',

                './templates_5.html',

                './templates_6.html',

                './templates_7.html',

                './templates_8.html',

                './templates_9.html',

                './templates_10.html'

            ];

            

            allTemplates = [];

            const parser = new DOMParser();


            // Progressively scan, parse, decode and compile all chunks present inside folders environment

            for (const file of filesToFetch) {

                try {

                    const response = await fetch(file);

                    if (!response.ok) continue; // Silently skip files that aren't created yet without throwing errors

                    

                    const htmlText = await response.text();

                    const doc = parser.parseFromString(htmlText, 'text/html');

                    const rawJsonText = doc.body.textContent.trim();

                    

                    if (!rawJsonText) continue;

                    

                    const parsedPayload = JSON.parse(rawJsonText);

                    if (parsedPayload && parsedPayload.result && parsedPayload.result.data && parsedPayload.result.data.json) {

                        const chunks = parsedPayload.result.data.json.map(item => ({

                            ...item,

                            cleanedBody: cleanBodyText(item.body)

                        }));

                        // Concatenate single files array chunks cleanly into main dataset engine

                        allTemplates = allTemplates.concat(chunks);

                    }

                } catch (err) {

                    console.log(`Note: Local storage file entry ${file} skipped or not yet initialized.`);

                }

            }


            // Sync visual components updates across systems mapping

            if (allTemplates.length > 0) {

                document.getElementById('stats-counter').innerText = `Successfully integrated ${allTemplates.length} library engines`;

                

                buildFormatFilters();

                renderTemplateDeck();

                selectTemplate(allTemplates[0].id);

            } else {

                document.getElementById('stats-counter').innerText = "System loading mismatch: Verify templates.html or your chunk data files exist.";

            }

        }


        // Generate categorical dynamic filter pill headers

        function buildFormatFilters() {

            const container = document.getElementById('format-filters');

            // Re-initialize containers clearly to prevent element duplication processing loops

            container.innerHTML = `<button class="filter-btn active bg-indigo-600 text-white px-3 py-1.5 rounded-xl text-xs font-medium transition-all cursor-pointer" data-format="all">All</button>`;

            

            const formats = [...new Set(allTemplates.map(t => t.format).filter(Boolean))];

            

            formats.forEach(format => {

                const btn = document.createElement('button');

                btn.className = "filter-btn bg-slate-100 text-slate-600 hover:bg-slate-200 px-3 py-1.5 rounded-xl text-xs font-medium transition-all cursor-pointer";

                btn.innerText = format.replace('_', ' ');

                btn.setAttribute('data-format', format);

                btn.onclick = () => {

                    document.querySelectorAll('.filter-btn').forEach(b => b.classList.remove('bg-indigo-600', 'text-white'));

                    btn.classList.add('bg-indigo-600', 'text-white');

                    activeFormat = format;

                    renderTemplateDeck();

                };

                container.appendChild(btn);

            });

            

            // Reassign base default baseline click handlers parameters

            document.querySelector('[data-format="all"]').onclick = function() {

                document.querySelectorAll('.filter-btn').forEach(b => b.classList.remove('bg-indigo-600', 'text-white'));

                this.classList.add('bg-indigo-600', 'text-white');

                activeFormat = 'all';

                renderTemplateDeck();

            };

        }


        // Render dynamic navigation view matching criteria selections

        function renderTemplateDeck() {

            const listContainer = document.getElementById('templates-list');

            const searchQuery = document.getElementById('search-input').value.toLowerCase();

            listContainer.innerHTML = '';


            const filtered = allTemplates.filter(t => {

                const matchesFormat = (activeFormat === 'all' || t.format === activeFormat);

                const matchesSearch = (t.name.toLowerCase().includes(searchQuery) || 

                                       t.description.toLowerCase().includes(searchQuery) ||

                                       t.cleanedBody.toLowerCase().includes(searchQuery));

                return matchesFormat && matchesSearch;

            });


            filtered.forEach(t => {

                const row = document.createElement('button');

                row.className = `w-full text-left p-3.5 transition-all hover:bg-slate-50 flex items-start gap-3 cursor-pointer border-b border-slate-50 ${activeTemplate && activeTemplate.id === t.id ? 'bg-indigo-50/60 hover:bg-indigo-50/60 border-l-4 border-l-indigo-600 pl-2.5' : ''}`;

                row.innerHTML = `

                    <span class="text-lg shrink-0">${t.emoji || '📄'}</span>

                    <div class="min-w-0 flex-1">

                        <h4 class="font-semibold text-sm text-slate-800 truncate">${t.name || 'Untitled Template'}</h4>

                        <p class="text-xs text-slate-400 truncate mt-0.5">${t.description || 'No summary parameters provided.'}</p>

                    </div>

                `;

                row.onclick = () => selectTemplate(t.id);

                listContainer.appendChild(row);

            });

        }


        // Focus interactive workspace controls on specific targeted dynamic component choice

        function selectTemplate(id) {

            activeTemplate = allTemplates.find(t => t.id === id);

            if (!activeTemplate) return;


            // Highlight row background selectors

            renderTemplateDeck();


            document.getElementById('active-emoji').innerText = activeTemplate.emoji || '📄';

            document.getElementById('active-title').innerText = activeTemplate.name || 'Untitled Template';

            document.getElementById('active-desc').innerText = activeTemplate.description || '';


            const container = document.getElementById('dynamic-fields-container');

            container.innerHTML = '';


            const fields = extractPlaceholders(activeTemplate.cleanedBody);

            

            if (fields.length === 0) {

                container.innerHTML = `<p class="text-xs text-slate-400 italic">Static architecture formula base pattern. Direct compile copy ready.</p>`;

            } else {

                fields.forEach(field => {

                    const block = document.createElement('div');

                    block.className = "flex flex-col gap-1.5";

                    block.innerHTML = `

                        <label class="text-xs font-semibold text-slate-600">${field}</label>

                        <input type="text" data-field="${field}" placeholder="Type text for [${field}]..." 

                               class="field-input px-3 py-2 border border-slate-200 rounded-xl text-sm focus:border-indigo-500 focus:ring-4 focus:ring-indigo-50 transition-all outline-none">

                    `;

                    container.appendChild(block);

                });

                

                // Bind keystroke triggers for real-time live preview updates

                document.querySelectorAll('.field-input').forEach(input => {

                    input.addEventListener('input', updateCompiledOutputPreview);

                });

            }


            updateCompiledOutputPreview();

        }


        // Refresh text rendering engine calculations parameters adjustments outputs

        function updateCompiledOutputPreview() {

            if (!activeTemplate) return;

            let resultText = activeTemplate.cleanedBody;

            

            document.querySelectorAll('.field-input').forEach(input => {

                const variableKey = input.getAttribute('data-field');

                const userValue = input.value.trim();

                

                const finalReplacementText = userValue ? userValue : `[${variableKey}]`;

                

                resultText = resultText.split(`{${variableKey}}`).join(finalReplacementText);

                resultText = resultText.split(`[${variableKey}]`).join(finalReplacementText);

            });


            document.getElementById('output-preview').innerText = resultText;

        }


        // Copy outputs buffer context straight into systemic OS clipboards vectors

        document.getElementById('copy-button').onclick = async function() {

            const textToCopy = document.getElementById('output-preview').innerText;

            try {

                await navigator.clipboard.writeText(textToCopy);

                const originalInner = this.innerHTML;

                this.innerHTML = `<i data-lucide="check" class="w-4 h-4 text-emerald-400"></i> Copied!`;

                this.classList.replace('bg-slate-800', 'bg-slate-900');

                lucide.createIcons();

                setTimeout(() => {

                    this.innerHTML = originalInner;

                    this.classList.replace('bg-slate-900', 'bg-slate-800');

                    lucide.createIcons();

                }, 2000);

            } catch (err) {

                alert("Clipboard allocation permissions denied. Select all and press Ctrl+C manually.");

            }

        };
